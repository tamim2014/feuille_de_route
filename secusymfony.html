						namespace App\Entity;
						use Doctrine\ORM\Mapping as ORM;
						
						#[ORM\Entity(repositoryClass: App\Repository\ListeRepository::class)]
						class Liste
						{
							#[ORM\Id]
							#[ORM\GeneratedValue]
							#[ORM\Column]
							private ?int $id = null;
							
							#[ORM\Column(length: 100)] // nombre de caracteres
							private ?string $nom = null;
							
							#[ORM\Column(length: 100)]
							private ?string $prenom = null;
							
							#[ORM\Column(length: 50)]
							private ?string $prefecture = null;
							
							// getters/setters...
						}
				

						// 2️⃣  LE REPOSITORY (ORM): src/Repository/ListeRepository.php
						// SELECT * FROM  liste WHERE prefecture=$p
						// Ici, on utilise QueryBuilder ORM, pas SQL
						
						public function findByPrefecture(string $p): array
						{
							return $this->createQueryBuilder('l')
							   ->where('l.prefecture = :p')
							   ->setParameter('p', $p)
							   ->orderBy('l.nom', 'ASC')
							   ->getQuery()
							   ->getResult();
						}
				

						// 3️⃣   LE CONTRÔLEUR (ORM): Il récupère les objets et les envoie au template.

						namespace App\Repository;
						
						use App\Entity\Liste;
						use Doctrine\Bundle\DoctrineBundle\Repository\ServiceEntityRepository;
						use Doctrine\Persistence\ManagerRegistry;
						
						class ListeRepository extends ServiceEntityRepository
						{
							public function __construct(ManagerRegistry $registry)
                            {
							    parent::__construct($registry, Liste::class);  
						    } 
							
							public function findByPrefecture(string $p): array
                            {
							    return $this->createQueryBuilder('l')
                                    ->where('l.prefecture = :p')								
                                    ->setParameter('p', $p)								
                                    ->getQuery()

						    }
						}
				

				        // 2️⃣  Le Contrôleur : il demande au Repository

							]);

⭐ 1) Pourquoi tout reste vertical même sans align-items ?

Parce que tu confonds deux axes :

✔ flex-direction(Axe principal)
→ row = horizontal
→ column = vertical

✔ align-items(axe secondaire)
→ stretch = étire verticalement (si row)
→ stretch = étire horizontalement (si column)

Donc :

👉 align-items ne décide PAS si les éléments sont en colonne ou en ligne.
👉 C’est flex-direction qui décide.

Et toi, tu n’as PAS mis flex-direction: column.
Donc Flexbox utilise sa valeur par défaut :

✅ flex-direction: row (axe principal)
✅ align-items: stretch étire verticalement (l’axe secondaire)
